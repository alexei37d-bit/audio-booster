target 'App' do
  project 'App.xcodeproj'

  # Capacitor
  pod 'Capacitor', :path => '../../node_modules/@capacitor/ios'
  pod 'CapacitorCordova', :path => '../../node_modules/@capacitor/ios'

  # Подключаем все плагины (если они есть)
  pod 'CapacitorCookies', :path => '../../node_modules/@capacitor/cookies'
  pod 'CapacitorHttp', :path => '../../node_modules/@capacitor/http'
  pod 'CapacitorSystemBars', :path => '../../node_modules/@capacitor/system-bars'
  pod 'CapacitorWebView', :path => '../../node_modules/@capacitor/web-view'

  # Плагины для уведомлений (если нужны)
  # pod 'CapacitorPushNotifications', :path => '../../node_modules/@capacitor/push-notifications'

  post_install do |installer|
    installer.pods_project.targets.each do |target|
      target.build_configurations.each do |config|
        config.build_settings['IPHONEOS_DEPLOYMENT_TARGET'] = '14.0'
        config.build_settings['ENABLE_BITCODE'] = 'NO'
      end
    end
  end
end
